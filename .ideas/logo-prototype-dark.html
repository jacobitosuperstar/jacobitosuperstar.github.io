# dependency_walker.py
import asyncio
from collections import defaultdict

async def resolve(pkg: str, seen: set) -> dict:
    if pkg in seen:
        return {}
    seen.add(pkg)
    deps = await fetch_deps(pkg)
    results = await asyncio.gather(*[resolve(d, seen) for d in deps])
    return {pkg: results}
